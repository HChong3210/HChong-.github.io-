#import "DFCSaleRemindReselectSaleVC.h"
#import "DFCCRMHttpNewManager.h"
#import "NSUserDefaults+SafeAccess.h"
#import "NSDictionary+SafeAccess.h"
#import "DFCProgressHUD.h"
#import "DFCRefreshTableVC+EmptyConfig.h"
#import "DFCMacros.h"
#import "DFCCRMSalersVO.h"
#import "UIView+SCCLayout.h"
#import "DFCColorFontLibrary.h"
#import "Masonry.h"
#import "UIViewController+DFCCustomBar.h"
#import "DFCSaleRemindNotificationVC.h"
#import "NSArray+SafeAccess.h"
#import "DFCLog.h"

#pragma mark - def

@interface DFCSaleRemindReselectSaleVC (){

    NSInteger _selectIndex;

    BOOL      _isNotification;
}

@property (nonatomic, strong) NSMutableArray *dataArray;
@property (nonatomic, strong) UIButton       *doneButton;
@property (nonatomic, strong) UIView         *topView;
@property (nonatomic, strong) UIButton       *selectButton;
@property (nonatomic, strong) UILabel        *titleLabel;
@property (nonatomic, strong) UILabel        *notifiLabel;
@property (nonatomic, strong) UIImageView    *arrowImageView;

@property (nonatomic, strong) UILabel        *sectionLabel;

@end

@implementation DFCSaleRemindReselectSaleVC

#pragma mark - Life Cycle
- (void)viewDidLoad{

    [super viewDidLoad];

    self.title = @"选择销售";
    _selectIndex = -1;
    _isNotification = YES;
    [self uiConfig];
    [self refreshHeader];
}

- (void)dealloc{


}

#pragma mark - Private Methon

- (void)uiConfig{

    [self addBarRightButtons:@[self.doneButton]];
    self.doneButton.enabled = NO;

    self.tbView.frame = CGRectMake(0, 107, [DFCMacros deviceSize].width, [DFCMacros deviceSize].height - 107);
    self.tbView.separatorStyle = UITableViewCellSeparatorStyleSingleLine;
    self.tbView.tableFooterView = [[UIView alloc] initWithFrame:CGRectZero];

    [self.view addSubview:self.topView];
    [self.topView addSubview:self.selectButton];
    [self.topView addSubview:self.sectionLabel];
    [self.selectButton addSubview:self.notifiLabel];
    [self.selectButton addSubview:self.titleLabel];
    [self.selectButton addSubview:self.arrowImageView];

    [self.topView mas_makeConstraints:^(MASConstraintMaker *make) {
        make.top.equalTo(self.view).offset(64);
        make.left.equalTo(self.view);
        make.right.equalTo(self.view);
        make.height.equalTo(@107);
    }];

    [self.selectButton mas_makeConstraints:^(MASConstraintMaker *make) {
        make.top.equalTo(self.topView).offset(16);
        make.left.equalTo(self.topView);
        make.right.equalTo(self.topView);
        make.height.equalTo(@44);
    }];

    [self.titleLabel mas_makeConstraints:^(MASConstraintMaker *make) {
        make.left.equalTo(self.selectButton).offset(16);
        make.centerY.equalTo(self.selectButton);
    }];

    [self.notifiLabel mas_makeConstraints:^(MASConstraintMaker *make) {
        make.right.equalTo(self.arrowImageView.mas_left).offset(-10);
        make.centerY.equalTo(self.selectButton);
    }];

    [self.arrowImageView mas_makeConstraints:^(MASConstraintMaker *make) {
        make.right.equalTo(self.selectButton).offset(-16);
        make.centerY.equalTo(self.selectButton);
        make.width.equalTo(@9.5);
        make.height.equalTo(@16);
    }];

    [self.sectionLabel mas_makeConstraints:^(MASConstraintMaker *make) {
        make.top.equalTo(self.selectButton.mas_bottom).offset(20);
        make.left.equalTo(self.view).offset(16);
    }];
}

#pragma mark - NetWork Request

- (void)loadNewData{

    [super loadNewData];

    [self requstSalerList];
}

- (void)requstSalerList{

    NSDictionary *parameter = @{
                                    @"shopCode": [[NSUserDefaults dfc_dictionaryForKey:DFC_User_Dic] dfc_stringForKey:@"store"]
                                };

    [DFCCRMHttpNewManager getShopSalersListWithParameter:parameter callBack:^(id data, DFCHttpStatus status) {

        [self endRefreshHeader];
        if (status == DFCHttpSuccess) {
            [self addEmptyDelegate:self requestSuccess:YES];
            self.dataArray = [DFCCRMSalersVO arrayOfModelsFromDictionaries:data error:nil];
            [self.tbView reloadData];
        }else if(status == DFCHttpFaild){

            [self addEmptyDelegate:self requestSuccess:NO];
            [DFCProgressHUD delayDisplayHUDWithType:SCProgressHudType_Error enableUserInteraction:YES msg:data];
        }
    }];
}

#pragma mark - Event Response

- (void)doneButtonAction{

    DFCCRMSalersVO *salersVO = [self.dataArray objectAtIndex:_selectIndex];

    NSMutableString *ids = [@"" mutableCopy];
    for (int i = 0; i<[_selectIdArray count]; i++) {
        if (i == [_selectIdArray count] - 1) {
            [ids appendString:FormatStr(@"%@",[_selectIdArray dfc_stringWithIndex:i])];
        }else{
            [ids appendString:FormatStr(@"%@,",[_selectIdArray dfc_stringWithIndex:i])];
        }
    }

    NSDictionary *parameter = @{
                                @"belongSales": salersVO.salerId,
                                @"ids":ids,
                                @"isProduceChance":@(_isNotification)
                                };

    [DFCProgressHUD show];
    [DFCCRMHttpNewManager changeToNewSalerWithParameter:parameter callBack:^(id data, DFCHttpStatus status) {

        [DFCProgressHUD dismiss];
        if (status == DFCHttpSuccess) {
            [self redistributionMsg:_selectIdArray.count isNotification:_isNotification];

            NSMutableArray *viewControllers = [[self.navigationController viewControllers] mutableCopy];
            NSInteger index = [viewControllers indexOfObject:self];
            if (index>=2) {
                [viewControllers removeObjectAtIndex:index - 1];
                [self.navigationController setViewControllers:viewControllers];
            }
            [self.navigationController popViewControllerAnimated:YES];
            dispatch_after(dispatch_time(DISPATCH_TIME_NOW, (int64_t)(0.5 * NSEC_PER_SEC)), dispatch_get_main_queue(), ^{
                [DFCProgressHUD showTextWithStatus:FormatStr(@"已分配%ld名客户给%@",[_selectIdArray count],salersVO.salerName)];
            });

        }else if(status == DFCHttpFaild){

            [DFCProgressHUD delayDisplayHUDWithType:SCProgressHudType_Error enableUserInteraction:YES msg:data];
        }
    }];
}

- (void)selectButtonAction{

    DFCSaleRemindNotificationVC *notificationVC = [[DFCSaleRemindNotificationVC alloc] init];
    notificationVC.isNotification = _isNotification;
    notificationVC.callBack = ^(BOOL isNotifi){
        if (isNotifi) {
            _isNotification = YES;
            self.notifiLabel.text = @"通知";
        }else{
            _isNotification = NO;
            self.notifiLabel.text = @"不通知";
        }
    };
    [self.navigationController pushViewController:notificationVC animated:YES];
}

//埋点
- (void)redistributionMsg:(NSInteger)count isNotification:(BOOL)isNotification{
    [DFCLog log:@"过期回访提醒列表－重新分配－点击完成-是否通知到销售新商机"];
}

#pragma mark - Delegate Or DataSource

- (NSInteger)tableView:(UITableView *)tableView numberOfRowsInSection:(NSInteger)section{

    return self.dataArray.count;
}

- (UITableViewCell *)tableView:(UITableView *)tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath{
    if (self.dataArray.count == 0) {
        return [self frameZeroCell];
    }

    static NSString *CellIdentifier = @"DFCSaleRemindReselectCell";
    UITableViewCell *cell = [tableView dequeueReusableCellWithIdentifier:CellIdentifier];
    if (!cell) {
        cell = [[UITableViewCell alloc] initWithStyle:UITableViewCellStyleSubtitle
                                             reuseIdentifier:CellIdentifier];
        if ([cell respondsToSelector:@selector(setLayoutMargins:)]) {
            cell.layoutMargins = UIEdgeInsetsZero;
        }
        cell.separatorInset = UIEdgeInsetsZero;
    }
    DFCCRMSalersVO *salersVO = [self.dataArray objectAtIndex:indexPath.row];
    cell.tintColor = [DFCColorFontLibrary DFC_COLOR_Primary];
    cell.textLabel.text = salersVO.salerName;
    cell.textLabel.font = [DFCColorFontLibrary DFC_FONT_P2];
    cell.selectionStyle = UITableViewCellSelectionStyleNone;

    if (indexPath.row == _selectIndex) {

        cell.textLabel.textColor = [DFCColorFontLibrary DFC_COLOR_Primary];
        cell.accessoryType = UITableViewCellAccessoryCheckmark;
    }else{

        cell.textLabel.textColor = [DFCColorFontLibrary DFC_COLOR_Black1];
        cell.accessoryType = UITableViewCellAccessoryNone;
    }

    return cell;
}


- (CGFloat)tableView:(UITableView *)tableView heightForRowAtIndexPath:(NSIndexPath *)indexPath{

    if (self.dataArray.count == 0) {
        return 0;
    }

    return 44;
}

- (void)tableView:(UITableView *)tableView didSelectRowAtIndexPath:(NSIndexPath *)indexPath{

    if (indexPath.row < [self.dataArray count]) {

        _selectIndex = indexPath.row;
        _doneButton.enabled = YES;
        [self.tbView reloadData];
    }
}

#pragma mark - Getter and Setter

- (UIView *)topView{

    if (!_topView) {

        _topView = [[UIView alloc] init];
        _topView.backgroundColor = [DFCColorFontLibrary DFC_COLOR_Gray3];
    }
    return _topView;
}

- (UIButton *)selectButton{

    if (!_selectButton) {

        _selectButton = [[UIButton alloc] init];
        [_selectButton addTarget:self action:@selector(selectButtonAction) forControlEvents:UIControlEventTouchUpInside];
        _selectButton.backgroundColor = [DFCColorFontLibrary DFC_COLOR_White1];
    }
    return _selectButton;
}


- (UILabel *)titleLabel{

    if (!_titleLabel) {

        _titleLabel = [[UILabel alloc] init];
        _titleLabel.font = [DFCColorFontLibrary DFC_FONT_P2];
        _titleLabel.textColor = [DFCColorFontLibrary DFC_COLOR_Black1];
        _titleLabel.text = @"通知到销售的新商机";
    }
    return _titleLabel;
}

- (UILabel *)notifiLabel{

    if (!_notifiLabel) {

        _notifiLabel = [[UILabel alloc] init];
        _notifiLabel.font = [DFCColorFontLibrary DFC_FONT_P2];
        _notifiLabel.textColor = [DFCColorFontLibrary DFC_COLOR_Black1];
        _notifiLabel.text = @"通知";
    }
    return _notifiLabel;
}

- (UIImageView *)arrowImageView{

    if (!_arrowImageView) {

        _arrowImageView = [[UIImageView alloc] init];
        [_arrowImageView setImage:[UIImage imageNamed:@"setting_entry"]];
    }
    return _arrowImageView;
}

- (UILabel *)sectionLabel{

    if (!_sectionLabel) {

        _sectionLabel = [[UILabel alloc] init];
        _sectionLabel.font = [DFCColorFontLibrary DFC_FONT_P2];
        _sectionLabel.textColor = [DFCColorFontLibrary DFC_COLOR_Black3];
        _sectionLabel.text = @"选择要通知的销售";
    }
    return _sectionLabel;
}

- (UIButton *)doneButton {
    if (!_doneButton) {
        _doneButton = [UIButton buttonWithType:UIButtonTypeCustom];
        _doneButton.backgroundColor = [UIColor clearColor];
        [_doneButton addTarget:self action:@selector(doneButtonAction) forControlEvents:UIControlEventTouchUpInside];
        _doneButton.contentHorizontalAlignment = UIControlContentHorizontalAlignmentRight;
        _doneButton.titleLabel.font = [DFCColorFontLibrary DFC_FONT_S2WithBold:YES];
        [_doneButton setTitle:@"完成" forState:UIControlStateNormal];
        [_doneButton setTitleColor:[DFCColorFontLibrary DFC_COLOR_C1] forState:UIControlStateNormal];
        [_doneButton setTitleColor:[DFCColorFontLibrary DFC_COLOR_Gray1] forState:UIControlStateDisabled];
        _doneButton.frame = CGRectMake(0, 0, 60, 44);
    }
    return _doneButton;
}

@end
